import os  # Import the os module for interacting with the operating system

ENVIRONMENT_SIZE = 60000  # Set the size of the environment

# camera parameters
CAMERA_TRANSFORMS = [[None, 0.0, 0.0], [None, 0.0, 90.0],]  # List of camera transforms; each contains [distance, azimuth, elevation]
CAMERA_TYPES = ['orthographic', 'orthographic', ]  # List of camera types corresponding to the transforms; options are 'orthographic' or 'perspective'
FACES_PER_PIXEL = 50  # Set the number of faces per pixel for the rendering process
IMAGE_SIZE = 512  # Set the size (resolution) of the rendered images

# object parameters
TARGET_IMAGES = ['data/cartoon1.jpg','data/cartoon2.jpg']  # List of target image file paths for silhouette matching
SOURCE_OBJECTS_PATH = "data/kitchen"  # Define the folder path containing source object files
SELECTED = os.listdir(SOURCE_OBJECTS_PATH)  # List all files in the source objects directory
SOURCE_OBJECTS = [os.path.join(SOURCE_OBJECTS_PATH, obj) for obj in SELECTED][:40]  # Create full paths for source objects and select the first 40 objects
SOURCE_SCALES = [50]*len(SOURCE_OBJECTS)  # Create a list of scales for source objects, each set to 50
RANDOM_INITIAL_TRANSFORMATIONS = True  # Enable random initial transformations for source objects

# training parameters
NUM_EPOCHS = 300  # Set the number of training epochs
WEIGHT_SILHOUETTE = 0.5  # Set the weight for the silhouette loss component
WEIGHT_INTERSECTION = 0.1  # Set the weight for the intersection loss component
WEIGHT_BOUNDING = 0.0001  # Set the weight for the bounding loss component
WEIGHT_LOG = 1.0  # Set the weight for the Log loss component
LEARNING_RATE = 1e-3  # Set the learning rate for the optimizer

# visualization parameters
PLOT_PERIOD = NUM_EPOCHS // 100  # Define how frequently to plot results during training (every PLOT_PERIOD epochs)
PROGRESSIVE_RESULTS = True  # Enable progressive visualization of results during training
VISUALIZE = False  # Disable detailed visualization during training

# derived hyperparameters
CAMERA_COUNT = len(CAMERA_TRANSFORMS)  # Calculate the number of cameras based on the length of the CAMERA_TRANSFORMS list
